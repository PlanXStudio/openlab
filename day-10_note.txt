int __io_putchar(int ch)
{
	if (ch == '\n') {
		HAL_UART_Transmit(&huart2, (uint8_t *)"\r", 1, 0xFFFF);
	}
	HAL_UART_Transmit(&huart2, (uint8_t *)&ch, 1, 0xFFFF);

	return ch;
}
